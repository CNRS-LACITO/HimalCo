\section{\ipa{mto}}
\textit{Vl}
\ipa{pɯ}
voir 
\zh{看见} 
\ipa{ɯ-mɲaʁ ɲɯ-mto ʂɯŋʂɯŋ ʑo}
\zh{他眼睛看得很清楚（视力很好）}
\subsection{\ipa{sɯmto}}
\textit{Vt}
\ipa{pɯ-}
montrer, laisser voir
\zh{让人看见、给人看}
\subsection{\ipa{ʑɣɤmto}}
\textit{Vi}
\ipa{pɯ-}
se voir
\zh{看到自己}
\ipa{tɯci ɯ-ŋgɯ ɕ-pɯ-ru-a ri, pjɯ-ntɕhar-a ɲɯ-ŋu tɕe pɯ-ʑɣɤmto-a}
\zh{我往水里看了一下，里面有我的倒影，我看到我自己了}
\subsection{\ipa{ʑɣɤsɯmto}}
textit{Vi}
\ipa{pɯ-}
se faire voir
\zh{让别人看到自己}
\ipa{ma-pɯ-tɯ-ʑɣɤsɯmto}
\zh{你不要让人看见你}
\ipa{ɯʑo mɯ-to-rɯndzaŋspa tɕe pjɤ-ʑɣɤsɯmto}
\zh{他不小心让人看见了}
\subsection{\ipa{amto}}
\textit{Vi}
être visible
\zh{看得见}
\ipa{khɯɣɲɟɯ ju-kɯ-ru tɕe, qhaqhu nɯra tɯrme nɯra amto}
\zh{往窗子外面看的话，看得见房子后面的人}
