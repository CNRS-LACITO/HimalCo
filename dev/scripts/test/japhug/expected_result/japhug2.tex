\section*{\ipa{sɤŋo}}
\textit{Vl}
\ipa{nɯ-}
\ipa{kɤ-}
écouter, ressentir
\zh{听、感受到}
\zh{做及物动词的时候，有“听从、服从”的意思}
\ipa{nɯ-sɤŋo-a}
\zh{我听了}
\ipa{kɤ-sɤŋo-t-a}
\zh{我听从他了}
\ipa{dianhua khro jɤ-lat-a ri, kɯ-sɤŋo maŋe}
\zh{我打了很多次电话，没有人接}
\ipa{wuma ɣɯ-sɤŋo-a}
\zh{他很听我的话}
\ipa{ŋgumdʑɯɣ ra kɯ ta-tɯt nɯ kɤ-sɤŋo-t-a}
\zh{他听从了领导的话}
\ipa{a-wa kɯ ta-tɯt nɯ kɤ-sɤŋo-t-a}
\zh{我听从了我父亲的话}
\ipa{tɤ-ti jɤɣ ma kɯre ku-sɤŋo-a}
\zh{你讲吧，我在这里听着}

\section*{\ipa{zjaŋzjaŋ}}
\textit{Ideo.2}
haut
\zh{身子高（比其他人高）}

\zh{他骑上了马，显得很高}
\subsection*{\ipa{zjaŋnɤzjaŋ}}
\textit{Ideo.3}
