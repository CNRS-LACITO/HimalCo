14  @param filename The name of the MDF file to read with full path, for instance 'user/input.txt'.
15  @param mdf2lmf A Python dictionary describing the mapping between MDF markers and LMF representation. Default value is 'mdf_lmf' dictionary defined in 'src/config/mdf.py'. Please refer to it as an example.
16  @param id A Python string identifying the created lexicon.
17  @return A Lexicon instance containing all lexical entries.
18  """
19  import re
20  mdf_file = open_read(filename, encoding='utf8')
21  # MDF syntax is the following: '\marker value'
22  mdf_pattern = """^\\\(\w*) (<(.*)>)? ?(.*)$"""
23  # Create a Lexicon instance to contain all lexical entries
24  lexicon = Lexicon(id)
25  # Add each lexical entry to the lexicon
26  current_entry = None
27  for line in mdf_file.readlines():
28  # Do not parse empty lines
29  if line != EOL:
30  result = re.match(mdf_pattern, line)
31  if result is None:
32  # Line is empty => continue parsing next line
33  continue
34  marker = result.group(1)
35  attrs = result.group(3)
36  value = result.group(4)
37  # Remove end-of-line characters
38  value = value.rstrip('\r\n')
39  # Do not consider empty fields
40  if value == "":
41  continue
42  # 'lx' marker indicates a new entry
43  if marker == "lx":
44  # Compute a unique identifier
45  occurrence_nb = 1
46  # Python equivalent of do...while loop
47  while True:
48  uid = value + "_" + str(occurrence_nb)
49  # If already taken, increment it
50  if lexicon.find_lexical_entries(lambda lexical_entry: lexical_entry.get_id() == uid) != []:
51  occurrence_nb += 1
52  else:
53  # Create a new entry
54  current_entry = LexicalEntry(uid)
55  break
56  # Add it to the lexicon
57  lexicon.add_lexical_entry(current_entry)
58  # Map MDF marker and value to LMF representation
59  try:
60  if attrs is not None:
61  # There are attributes
62  attributes = {}
63  # Remove quotation marks from attributes if any
64  attrs = attrs.replace('"', '')
