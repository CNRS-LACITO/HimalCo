71 def format_lexeme(lexical_entry, font):
72  """! @brief 'lx', 'hm' and 'lc' fields are flipped if 'lc' field has data.
73  @param lexical_entry The current Lexical Entry LMF instance.
74  @param font A Python dictionary giving the vernacular, national, regional fonts to apply to a text in LaTeX format.
75  @return A string representing lexeme in LaTeX format.
76  """
77  lexeme = font[VERNACULAR](lexical_entry.get_lexeme())
78  result = "\\vspace{1cm} \\hspace{-1cm} "
79  if lexical_entry.get_homonymNumber() is not None:
80  # Add homonym number to lexeme
81  lexeme += " \\textsubscript{" + str(lexical_entry.get_homonymNumber()) + "}"
82  if lexical_entry.get_contextual_variations() is not None and len(lexical_entry.get_contextual_variations()) != 0:
83  # Format contextual variations
84  for var in lexical_entry.get_contextual_variations():
85  result += " " + font[VERNACULAR](var)
86  result += " (from: " + lexeme + ")."
87  else:
88  # Format lexeme
89  result += lexeme
90  result += " \\hspace{0.2cm} \\hypertarget{" + unicode(lexical_entry.get_id()).replace('_', '\_') + "}{}" + EOL
91  return result
