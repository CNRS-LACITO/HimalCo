765  """! @brief Set paradigm.

769  @param anymacy Anymacy, e.g. animate or inanimate.

788  word_form.set_anymacy(anymacy)
789  if grammatical_number is not None:
790  word_form.set_grammaticalNumber(grammatical_number)
